exports.handler = async function(event) { if (event.httpMethod === "OPTIONS") { return { statusCode: 200, headers: corsHeaders(), body: "" }; } if (event.httpMethod !== "POST") { return json(405, { error: "Method not allowed" }); } try { const apiKey = process.env.DEEPSEEK_API_KEY; if (!apiKey) { return json(500, { error: "Не найден DEEPSEEK_API_KEY в переменных окружения Netlify" }); } const body = JSON.parse(event.body || "{}"); const { scenario, stepId, question, expected, userAnswer, history = [] } = body; if (!question || !expected || !userAnswer) { return json(400, { error: "Не хватает question, expected или userAnswer" }); } const systemPrompt = ` Ты AI XR Instructor для экзамена по первой помощи и СЛР. Твоя задача: 1. Оценить смысл ответа ученика. 2. Не требовать дословного совпадения. 3. Если ответ безопасный и по сути правильный — correct=true. 4. Если ученик пропустил критически важный шаг — correct=false. 5. Дать короткую обратную связь на русском. 6. Дать следующий вопрос, если ответ правильный. 7. Не шути. Не добавляй лишнее. Стиль — спокойный инструктор. Отвечай строго JSON: { "correct": true или false, "feedback": "короткая обратная связь", "nextQuestion": "следующий вопрос инструкторa" } `; const userPrompt = ` Сценарий: ${scenario} Текущий этап: ${stepId} Вопрос инструктора: ${question} Ожидаемый смысл правильного ответа: ${expected} Ответ ученика: ${userAnswer} История последних ответов: ${JSON.stringify(history, null, 2)} Оцени ответ. `; const response = await fetch("https://api.deepseek.com/chat/completions", { method: "POST", headers: { "Content-Type": "application/json", "Authorization": `Bearer ${apiKey}` }, body: JSON.stringify({ model: "deepseek-v4-flash", messages: [ { role: "system", content: systemPrompt }, { role: "user", content: userPrompt } ], thinking: { type: "disabled" }, response_format: { type: "json_object" }, temperature: 0.2, max_tokens: 500, stream: false }) }); const aiRaw = await response.json(); if (!response.ok) { return json(response.status, { error: "Ошибка DeepSeek API", details: aiRaw }); } const content = aiRaw?.choices?.[0]?.message?.content; if (!content) { return json(500, { error: "DeepSeek не вернул content", details: aiRaw }); } let parsed; try { parsed = JSON.parse(content); } catch (e) { return json(500, { error: "AI вернул невалидный JSON", raw: content }); } return json(200, { correct: Boolean(parsed.correct), feedback: parsed.feedback || "Ответ принят.", nextQuestion: parsed.nextQuestion || "" }); } catch (error) { return json(500, { error: error.message || "Server error" }); } }; function json(statusCode, payload) { return { statusCode, headers: { ...corsHeaders(), "Content-Type": "application/json; charset=utf-8" }, body: JSON.stringify(payload) }; } function corsHeaders() { return { "Access-Control-Allow-Origin": "*", "Access-Control-Allow-Headers": "Content-Type, Authorization", "Access-Control-Allow-Methods": "POST, OPTIONS" }; }
